Wybraem LeaveOneOut Encoding oraz CountEncoding. Pierwszy z nich dziaa podobnie jak Target Encoding, z t rnic, e przy wyliczaniu redniej nie bierze pod uwag danego rekordu. Dziaa tylko, jeeli zmienna celu jest typu binarnego albo cigego. Natomiast CountEncoding zamienia nazw kategorii na liczb jej wystpie, dlatego nie ma adnych ogranicze co do typu zmiennej celu.

from sklearn.impute import KNNImputer

remove_n = int(0.1 * len(allegro_df_mod))

for r in range (len(rmse)):
    allegro_tmp = allegro_df_mod.copy()
    r_temp = 0
    drop_indices = np.array(np.random.choice(allegro_df_mod.index, remove_n, replace=False))

    for i in drop_indices:
        allegro_tmp.loc[i, 'it_seller_rating'] = np.NaN

    imputer = KNNImputer(n_neighbors=3, weights="uniform")
    NN = pd.DataFrame(imputer.fit_transform(allegro_tmp), columns = cols)

    for i in range (len(NN)):
        r_temp += (sr - NN.loc[i, 'it_seller_rating'])**2
    r_temp /= len(NN)
    r_temp = np.sqrt(r_temp).round(2)

    rmse[r] = r_temp
print(rmse)
std_r = np.std(rmse)
mean_r = np.mean(rmse)
