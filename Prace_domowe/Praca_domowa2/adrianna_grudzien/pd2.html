Metoda helmert encoding porwnuje poziomy zmiennej ze redni kolejnych poziomw zmiennej - kady poziom zmiennej jest porwnywany z pniejszym poziomem zmiennych. Ten typ kodowania jest przydatny, gdy poziomy zmiennej kategorycznej s uporzdkowane w znaczcy sposb. W naszym przypadku nie ma takie sytuacji, ale warto byo si przyjrze.

Jedn z wad tej metody jest dua liczba nowopowstaych kolumn, co w przypadku liczby zmiennych wikszych ni 4, staje si mao efektywnie pamiciowo.

# uzupeniamy przy uyciu narzdzia Nearest neighbors imputation o parametrze n_neighbors=3
def uzupelnij_dane(df_nan):
    df_imputed = df_nan.copy(deep=True)
    imputer = KNNImputer(n_neighbors=3, weights='uniform')
    imputed = imputer.fit_transform(df_imputed)
    df_imputed = pd.DataFrame(imputed, columns=df_imputed.columns)
    return df_imputed

df_imputed = uzupelnij_dane(df_nan)

# liczymy bd redniokwadratowy (RMSE):
def licz_blad(df_numeric, df_nan, df_imputed, nazwa_kolumny):
    nan_indexes = df_numeric[nazwa_kolumny].index[df_nan[nazwa_kolumny].apply(np.isnan)]
    real_data = df_numeric[nazwa_kolumny][nan_indexes]
    fake_data = df_imputed[nazwa_kolumny][nan_indexes]
    return np.sqrt(rmse(real_data, fake_data))
