def chose_components(n, X, score_fun):
    scores = []  
    for i in range(2, n):
        gmm = GMM(n_components = i, covariance_type = 'full')
        labels = gmm.fit_predict(X)
        wcss = score_fun(X, labels)
        scores.append(wcss)
    plt.plot(range(2, n), scores)
    return scores
